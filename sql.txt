select DATE, PLACE, IDS.ID from ((
  (( SELECT s1.ID, s1.FLIGHT_ID, s1.PLACE from SEAT s2, SEAT s1 WHERE s1.OCCUPIED = FALSE and s1.ID = s2.ID-1 and s2.OCCUPIED = TRUE ))
union
( (SELECT s1.ID, s1.FLIGHT_ID, s1.PLACE from SEAT s1, SEAT s2 WHERE s1.OCCUPIED = FALSE and s1.ID-1 = s2.ID and s2.OCCUPIED = TRUE))
)
ORDER BY ID) AS IDS
INNER JOIN FLIGHT ON IDS.FLIGHT_ID = FLIGHT.ID
ORDER BY DATE, PLACE
